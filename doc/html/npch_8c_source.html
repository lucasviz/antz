420 //
421 // Must run in another thread, reads data into circular buffer
422 //
423 // Currently have two options for behavior when we reach the end:
424 // - pause
425 // - loop
426 //
427 // If threads are supported this function will be called in its own thread.
428 // In that case, it will implement circular buffer as part of loading data --
429 // waiting for data to be read before overwriting it.
430 // With looping with files larger than the circular buffer size, the large files
431 // are re-opened and re-read. This allows looping to occur with any size file.
432 //
433 // If threads are not supported, the buffer loading stops as soon as the buffer
434 // is full. The remaining data is never loaded.
435 // Looping occurs only on the data loaded into the buffer.
436 //------------------------------------------------------------------------------
437 bool threadRunning = false;
438 void npCircularBufferLoadLoop (void* dataRef)
439 {
440  pData data = (pData)dataRef;
441  pNPch ch = &data->io.ch;
442  pNPtrackDataSource trackDataSource = &ch->trackDataSource;
443  char** tokens = NULL;
444 
445  ch->numberTracks = SaveHeaders ( npChReadTrackTokens(ch, data), data);
446  CreateTracks (ch->numberTracks, data);
447 
448  // reset read/write index on reload
449  ch->channelReadIndex = kNPtrackBufferSize - 1;
450  ch->channelWriteIndex = 0;
451  while ( npChHasValidDataSource( trackDataSource ))
452  {
453  while ( true )
454  {
455  tokens = npChReadTrackTokens(ch, data);
456  if (tokens == NULL) break;
457 
458  npChSaveTrackData( ch->numberTracks, ch->channelWriteIndex++,
459  ch->trackFloatData, ch->trackUcharData, ch->trackIntData, tokens );
460  if (ch->channelWriteIndex > ch->channelDataSize)
461  ch->channelDataSize = ch->channelWriteIndex;
462 
463  // circular buffer requires thread, otherwise, we are totally done
464  if (nposSupportsAntzThreads())
465  {
466  if (ch->channelWriteIndex == kNPtrackBufferSize)
467  ch->channelWriteIndex = 0;
468  //printf("channelReadIndex %d, channelWriteIndex %d, channelDataSize now %d\n", data->io.ch.channelReadIndex, data->io.ch.channelWriteIndex, data->io.ch.channelDataSize);
469 
470  // wait for data to be read before overwriting
471  while (ch->channelWriteIndex == ch->channelReadIndex)
472  {
473  //printf ("Waiting for data to be read..\n");
474  nposSleep(0.1);
475  }
476  } // without threads, break out
477  else if (ch->channelWriteIndex == kNPtrackBufferSize)
478  {
479  break;
480  }
481  }
482  npCloseDataSource (trackDataSource, dataRef);
483  //printf("ALL DONE!\n");
484  // if we are looping, and the file was too large to fit entirely into track memory,
485  // re-read the file
486  if ( loopAtEndOfTrack
487  && ch->channelDataSize == kNPtrackBufferSize
488 //JJ-zz && nposSupportsAntzThreads()
489  && (ch->trackDataSource.dataSourceType == kDataFromFile))
490  {
491  printf("LOOPING AT END OF TRACK\n");
492  data->io.ch.channelReadIndex = kNPtrackBufferSize - 1;
493  data->io.ch.channelWriteIndex = 0;
494  npReopenDataSource( ch, dataRef );
495  }
496  }
497  threadRunning = false;
498  npPostMsg ("..npCircularBufferLoadLoop exit", kNPmsgCtrl, data);
499  nposEndThread();
500 }
